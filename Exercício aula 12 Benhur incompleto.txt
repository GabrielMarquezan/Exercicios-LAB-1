int main()
{
   int vetor1[5], vetor2[5], vetor3[10];
   
   for(int i = 0; i < 5; i++)
   {
       printf("Digite um número: ");
       scanf("%d", &vetor1[i]);
       
       if(i == 4)
       {
           printf("\n");
           
           for(int i = 0; i < 5; i++)
           {
               printf("Posição %d do vetor 1: %d\n", i, vetor1[i]);
           }
           
           printf("\n");
       }
   }

   for(int i = 0; i < 5; i++)
   {
       printf("Digite um número: ");
       scanf("%d", &vetor2[i]);
       
       if(i == 4)
       {
           printf("\n");
           
           for(int i = 0; i < 5; i++)
           {
               printf("Posição %d do vetor 2: %d\n", i, vetor2[i]);
           }
       
           printf("\n");
       }
   }
   
   for(int i = 0; i < 5; i++)
   {
       vetor3[i] = vetor1[i];

       {
          for(int i = 0; i < 10; i++)

              printf("Posição %d do vetor 3: %d\n", i, vetor3[i]);

}
