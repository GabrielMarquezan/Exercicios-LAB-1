#include <stdlib.h>
#include <stdio.h>

void probabilidade(void);

int main()
{
    probabilidade();
}
    
int dado(void)
{
    return rand() % 6 + 1;
}

void probabilidade(void)
{
    int contador[6];
    int resultado;
    
    for(int i = 0; i < 100000; i++)
    {
        resultado = dado();
        
        if(resultado == 1)
        {
            contador[0]++;
        }
        else if(resultado == 2)
        {
            contador[1]++;
        }
        else if(resultado == 3)
        {
            contador[2]++;
        }
        else if(resultado == 4)
        {
            contador[3]++;
        }
        else if(resultado == 5)
        {
            contador[4]++;
        }
        else
        {
            contador[5]++;
        }
    }
    
    for(int i = 0; i < 6; i++)
    {
        float chance = contador[i] / 1000;
        printf("%d foi o resultado %d vezes\n", i + 1, contador[i]);
        printf("Sua probabilidade é de %.2f%%\n", chance);
    }
}

#include <stdlib.h>
#include <stdio.h>

void probabilidade(void);

int main()
{
    probabilidade();
}
    
int dado(void)
{
    return rand() % 6 + 1;
}

void probabilidade(void)
{
    int contador[6];
    int resultado;
    int teste = 100000;
    float desonestidade;
    
    for(int i = 0; i < teste; i++)
    {
        resultado = dado();
        
        if(resultado == 1)
        {
            contador[0]++;
        }
        else if(resultado == 2)
        {
            contador[1]++;
        }
        else if(resultado == 3)
        {
            contador[2]++;
        }
        else if(resultado == 4)
        {
            contador[3]++;
        }
        else if(resultado == 5)
        {
            contador[4]++;
        }
        else
        {
            contador[5]++;
        }
    }
    
    for(int i = 0; i < 5; i++)
    {
        int aux;
        for(int j = 1; j < 6; j++)
        {
            if(contador[i] > contador[j])
            {
                aux = contador[i];
                contador[i] = contador[j];
                contador[j] = aux;
            }
        }
    }
    
    desonestidade = (contador[5] - contador[0]) / teste;
    
    printf("O Fator de Desonestidade é: %f\n", desonestidade);
}

#include <stdio.h>

{
    int vetor[100];
    int maior = 0;
    int menor = 2147483647;
    
    for(int i = 0; i < 100; i++)
    {
        vetor[i] = rand();
        
        if(vetor[i] > maior)
        {
            maior = vetor[i];
        }
        
        if(vetor[i] < menor)
        {
            menor = vetor[i];
        }
    }
    
    printf("o menor número é: %d\n", menor); 
    printf("o maior número é: %d\n", maior); 
}
