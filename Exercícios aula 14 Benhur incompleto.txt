//Criar strlen

int tamanho_str(char string[])
{
    int tamanho = 0;

    while(string[tamanho] != '\0')
    {
        tamanho++;
    }

    return tamanho;
}

//Criar strcpy

void copia_str(char destino[], char origem[])
{
    int tamanho = 0;

    while(destino[tamanho] != '\0')
    {
        tamanho++;
    }

    for(int i = 0; i <= tamanho; i++)
    {
        if(destino[i] == '\0')
        {
            destino[i + 1] = '\0';
        }

        destino[i] = origem[i];
    }
}

//Criar strcat

void junta_str(char primeira[], char segunda[])
{
    int index1 = 0;
    int index2 = 0;

    while(true)
    {
        if(primeira[index1] == '\0')
        {
            while(true)
            {
                primeira[index1] = segunda[index2];

                if(segunda[index2] == '\0')
                {
                    break;
                }

                index1++;
                index2++;
            }

            break;
        }

        index1++;
    }
}

bool palindromo(char string[])
{
    int tamanho = 0;
    bool resultado = true;

    while(string[tamanho] != '\0')
    {
        tamanho++;
    }
    
    int aux = tamanho - 1;

    for(int i = 0; i < tamanho / 2; i++)
    {
        if(string[i] != string[aux])
        {
            resultado = false;
        }

        aux--;
    }

    return resultado;
}
